<파일 시스템 생성 후 볼륨 마운트>
mkfs -t ext4 /dev/nvme1n1
cd ~
mkdir temp
mount /dev/nvme1n1 temp/
df -h


<vi 편집기를 사용하여 test.txt 생성>
vi temp/test.txt
ls -al temp/


<언마운트 후 파일 시스템 등록 해제>
umount temp
df -h


<temp 폴더에 볼륨 마운트 후 test.txt 파일 생성>
sudo -s
cd ~
mkdir temp
mount /dev/nvme1n1 temp/
df -h
ls temp/
cat temp/test.txt


<EFS 마운트 후 test.txt 파일 생성>
cd ~
pwd
mkdir efs
sudo mount –t nfs4 … //EFS에서 복사했던 명령어
ls -al
sudo chown ec2-user:ec2-user efs
ls -al efs/
vi efs/test.txt
cat efs/test.txt
ls efs/


<EFS 마운트 후 test.txt 파일 확인, 그 후 언마운트>
cd ~
pwd
mkdir efs
sudo mount –t nfs4 … //EFS에서 복사했던 명령어
ls -al efs/
cat efs/test.txt
sudo rm efs/test.txt
sudo umount efs
