<test_gemini.py 파일에 다음을 입력>
from google import genai
from dotenv import load_dotenv
import os

load_dotenv()

client = genai.Client(api_key=os.environ['GEMINI_API_KEY'])
response = client.models.generate_content(
    model="gemini-2.5-flash-lite",
    contents="안녕! 넌 누구야?"
)
print(response.text)


<test_gemini.py 실행>
